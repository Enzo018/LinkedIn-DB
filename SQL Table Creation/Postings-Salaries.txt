CREATE TABLE IF NOT EXISTS public.postings_salaries
(
    job_id bigint NOT NULL,
    salary_id integer NOT NULL,
    CONSTRAINT postings_salaries_pkey PRIMARY KEY (job_id, salary_id),
    CONSTRAINT postings_salaries_ref_posting_fk FOREIGN KEY (job_id)
        REFERENCES public.postings (job_id) MATCH SIMPLE
        ON UPDATE RESTRICT
        ON DELETE RESTRICT
        NOT VALID,
    CONSTRAINT postings_salaries_ref_salaries_fk FOREIGN KEY (salary_id)
        REFERENCES public.salaries (salary_id) MATCH SIMPLE
        ON UPDATE RESTRICT
        ON DELETE RESTRICT
        NOT VALID
)

TABLESPACE pg_default;

CREATE INDEX IF NOT EXISTS fki_postings_salaries_ref_postings_fk
    ON public.postings_salaries USING btree
    (job_id ASC NULLS LAST)
    TABLESPACE pg_default;
