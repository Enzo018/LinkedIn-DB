CREATE TABLE IF NOT EXISTS public.postings_skills
(
    job_id bigint NOT NULL,
    skill_abr character varying(20) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT postings_skills_pkey PRIMARY KEY (job_id, skill_abr),
    CONSTRAINT postings_skills_ref_postings_fk FOREIGN KEY (job_id)
        REFERENCES public.postings (job_id) MATCH SIMPLE
        ON UPDATE RESTRICT
        ON DELETE RESTRICT
        NOT VALID,
    CONSTRAINT postings_skills_ref_skills_fk FOREIGN KEY (skill_abr)
        REFERENCES public.skills (skill_abr) MATCH SIMPLE
        ON UPDATE RESTRICT
        ON DELETE RESTRICT
        NOT VALID
)

TABLESPACE pg_default;

CREATE INDEX IF NOT EXISTS fki_postings_skills_ref_postings_fk
    ON public.postings_skills USING btree
    (job_id ASC NULLS LAST)
    TABLESPACE pg_default;
-- Index: fki_postings_skills_ref_skills_fk

-- DROP INDEX IF EXISTS public.fki_postings_skills_ref_skills_fk;

CREATE INDEX IF NOT EXISTS fki_postings_skills_ref_skills_fk
    ON public.postings_skills USING btree
    (skill_abr COLLATE pg_catalog."default" ASC NULLS LAST)
    TABLESPACE pg_default;
