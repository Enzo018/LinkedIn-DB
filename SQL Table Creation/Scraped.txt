CREATE TABLE IF NOT EXISTS public.scraped
(
    job_posting_url character varying(1000) COLLATE pg_catalog."default" NOT NULL,
    scraped bigint,
    closed_time character varying(100) COLLATE pg_catalog."default",
    expiry character varying(100) COLLATE pg_catalog."default",
    CONSTRAINT scraped_pkey PRIMARY KEY (job_posting_url)
)

TABLESPACE pg_default;
